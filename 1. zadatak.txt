/*Napišite program u kojem inicijalizirate niz od 10 imena.
    Program ispisuje sva imena koja imaju prvo i zadnje slovo jednako.*/

#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#define N 7

int main() {
	char imena[N][15] = { "ana", "anita", "anamarija", "marko", "pero", "ivan", "oto" };

	for (int i = 0; i < N; i++)
	{
		if (imena[i][0] == imena[i][strlen(imena[i])-1])
		{
			printf("%s \n", imena[i]);
		}
	}
	system("pause");
}
